583                 /**
584                  * 
585                  */
586                 private static final long serialVersionUID = 565842888109816432L;

588                 public void actionPerformed(ActionEvent evt) {
589                     chooserKeyFile.setDialogType(JFileChooser.SAVE_DIALOG);
590                     int response = chooserKeyFile.showSaveDialog(rootPane);
591                     if (response == JFileChooser.APPROVE_OPTION) {
592                         File file = chooserKeyFile.getSelectedFile();
593                         if (!filterKey.accept(file)) {
594                             if (WaarpPassword.desModel) {
595                                 file = new File(file.getAbsoluteFile() + "." + Des.EXTENSION);
596                             } else {
597                                 file = new File(file.getAbsoluteFile() + "." + Blowfish.EXTENSION);
598                             }
599                         }
600                         jTextFieldKeyFile.setText(file.getAbsolutePath());
601                         try {
602                             waarpPassword.saveKey(file);
603                         } catch (CryptoException e) {
604                             // TODO Auto-generated catch block
605                             e.printStackTrace();
606                         }
607                         passwordModified = false;
608                     }
609                 }
610             };
611         }
612         return saveAsKeyAction;
613     }
